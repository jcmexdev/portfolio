func (o *Orchestrator) rollback(ctx context.Context, steps []Step) {
    for i := len(steps) - 1; i >= 0; i-- {
        step := steps[i]
        if err := step.Compensate(ctx); err != nil {
            slog.ErrorContext(ctx, "CRITICAL: compensation failed", "step", step.Name())
        }
    }
}
